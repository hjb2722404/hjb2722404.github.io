    # git 命令大全

    # 代码库

    # 在当前目录新建一个Git代码库

    $ git init
    1

    # 新建一个目录，将其初始化为Git代码库

    $ git init [project-name]
    1

    # 下载一个项目和它的整个代码历史

      $ git clone [url]
    1

    # 配置

    # 显示当前的Git配置

      $ git config --list
    1

    # 编辑Git配置文件

      $ git config -e [--global]
    1

    # 设置提交代码时的用户信息

      $ git config [--global] user.name "[name]"

  $ git config [--global] user.email "[email address]"
    1
    2
    3

    # 本地增删改文件

    # 添加指定文件到暂存区

      $ git add [file1] [file2] ...
    1

    # 添加指定目录到暂存区，包括子目录

      $ git add [dir]
    1

    # 添加当前目录的所有文件到暂存区

      $ git add .
    1

    # 添加每个变化前，都会要求确认,对于同一个文件的多处变化，可以实现分次提交

      $ git add -p
    1

    # 删除工作区文件，并且将这次删除放入暂存区

      $ git rm [file1] [file2] ...
    1

    # 停止追踪指定文件，但该文件会保留在工作区

      $ git rm --cached [file]
    1

    # 改名文件，并且将这个改名放入暂存区

      $ git mv [file-original] [file-renamed]
    1

    # 代码提交

    # 提交暂存区到仓库区

    $ git commit -m [message]
    1

    # 提交暂存区的指定文件到仓库区

    $ git commit [file1][file2]... -m [message]
    1

    # 提交工作区自上次commit之后的变化，直接到仓库区

    $ git commit -a
    1

    # 提交时显示所有diff信息

    $ git commit -v
    1

    # 使用一次新的commit，替代上一次提交，如果代码没有任何新变化，则用来改写上一次commit的提交信息

    $ git commit --amend -m [message]
    1

    # 重做上一次commit，并包括指定文件的变化

    $ git commit --amend [file1] [file2] ...
    1

    # 分支操作

    # 列出所有本地分支

    $ git branch
    1

    # 列出所有远程分支

    $ git branch -r
    1

    # 列出所有本地分支和远程分支

    $ git branch -a
    1

    # 新建一个分支，但依然停留在当前分支

    $ git branch [branch-name]
    1

    # 新建一个分支，并切换到该分支

    $ git checkout -b [branch]
    1

    # 新建一个分支，指向特定commit

    $ git branch [branch] [commit]
    1

    # 新建一个分支，与指定的远程分支建立追踪关系

    $ git branch --track [branch] [remote-branch]
    1

    # 切换到指定分支，并更新工作区

    $ git checkout [branch-name]
    1

    # 建立追踪关系，在现有分支与指定的远程分支之间

    $ git branch --set-upstream [branch] [remote-branch]
    1

    # 合并指定分支到当前分支

    $ git merge [branch]
    1

    # 选择一个commit，合并进当前分支

    $ git cherry-pick [commit]
    1

    # 删除分支

    $ git branch -d [branch-name]
    1

    # 批量删除本地分支(排除分支名包含master和dev的分支)

    $ git branch | grep -v -E 'master|dev' | xargs git branch -D
    1

    # 删除远程分支

    $ git push origin --delete [branch-name]
$ git branch -dr [remote/branch]
    1
    2

    # 批量删除远程分支(排除分支名包含master和dev的分支)

    $ git branch -r | grep -v -E `master|dev` | sed 's/origin\///g' | xargs -I {} git push origin : {}
    1

    如果有些分支无法删除，是因为远程分支的缓存问题，可以使用git remote prune

    $ git tag [tag]

    $ git tag [tag] [commit]

    # 查看tag信息

    $ git show [tag]

    # 提交指定tag

    $ git push [remote] [tag]
    1

    # 提交所有tag

    $ git push [remote] --tags
    1

    # 新建一个分支，指向某个tag

    $ git checkout -b [branch] [tag]
    1

    # 批量删除本地tag

    $ git tag | xargs -I {} git tag -d {}
    1

    # 批量删除远程tag

    $ git tag | xargs -I {} git push origin :refs/tags/{}
    1

    # 查看信息

    # 显示有变更的文件

    $ git status
    1

    # 显示当前分支的版本历史

    $ git log
    1

    # 显示提交历史，以及每次提交发生变更的文件

    $ git log --stat
    1

    # 显示某个文件的版本历史，包括文件改名

    $ git log --follow [file]
$ git whatchanged [file]
    1
    2

    # 显示指定文件相关的每一次diff

    $ git log -p [file]
    1

    # 显示最近的n次提交的信息

    $ git log -[n]
    1

    # 显示指定文件是什么人在什么时间修改过

    $ git blame [file]
    1

    # 显示暂存区和工作区差异

    $ git diff
    1

    # 显示暂存区和上一个提交的差异

    $ git diff --cached [file]
    1

    # 显示工作区与当前分支最新提交之间的差异

    $ git diff HEAD
    1

    # 显示两次提交之间的差异

    $ git diff [first-commit]...[last-commit]
    1

    # 仅比较统计信息

    $ git diff --stat
    1

    # 显示某次提交的元数据和内容变化

    $ git show [commit]
    1

    # 显示某次提交发生变化的文件

    $ git show --name-only [commit]
    1

    # 显示某次提交时，某个文件的内容

    $ git show [commit]:[filename]
    1

    # 显示最近的git操作日志

    $ git reflog
    1

    # 显示Git命令的帮助信息

    $ git help [command]
    1

    # 远程同步

    # 下载远程仓库的所有变动

    $ git fetch [remote]
    1

    # 显示所有远程仓库

    $ git remote -v
    1

    # 显示某个远程仓库的信息

    $ git remote show [remote]
    1

    # 增加一个新的远程仓库，并命名

    $ git remote add [shortname] [url]
    1

    # 取回远程仓库的变化，并与本地分支合并

    $ git pull [reomte] [branch]
    1

    # 上传本地指定分支到远程仓库

    $ git push [remote] [branch]
    1

    # 强行推送当前分支到远程仓库，即使有冲突

    $ git push [remote] --force
    1

    # 推送所有分支到远程仓库

    $ git push [remote] --all
    1

    # 贮藏操作

    # 将工作区已追踪的更改和暂存区贮藏到【栈】中

    $ git stash
    1

    # 列出【栈】中的所有stash

    $ git stash list
    1

    # 将stash恢复到暂存区和工作目录，但不删除stash

    $ git stash apply stash@{id}
    1

    # 将【栈】顶的stash恢复暂存区和工作目录，并且将这个stash从【栈】中删除

    $ git stash pop
    1

    # 删除【栈】中所有stash

    $ git stash drop stash@{id}
    1

    # 贮藏时包含未跟踪文件

    $ git stash -u
    1

    # 贮藏时包含所有文件（包括忽略的文件）[不推荐]

    $ git stash -a
    1

    # 贮藏时添加一个说明

    $ git stash save "说明"
    1

    # 查看指定stash的内容

    $ git stash show stash@{id}
    1

    # 撤销操作

    # 恢复暂存区的指定文件到工作区

    $ git checkout [file]
    1

    # 恢复某个commit的指定文件到工作区

    $ git checkout [commit] [file]
    1

    # 恢复上一个commit的所有文件到工作区

    $ git checkout .
    1

    # 重置暂存区的指定文件，与上一次commit保持一致，但工作区不变

    $ git reset [file]
    1

    # 将指定提交的更改恢复到暂存区，工作区不变

    $ git reset --soft [commit]
    1

    # 将指定提交的更改恢复到暂存区和工作目录

    $ git reset --mixed [commit]
    1

    # 重置暂存区与工作区，与上一次commit保持一致

    $ git reset --hard
    1

    # 重置当前分支的指针为指定commit，同时重置暂存区，但工作区不变

    $ git reset [commit]
    1

    # 重置当前分支的指针的HEAD为指定commit， 同时重置暂存区和工作区，与指定commit一致

    $ git reset --hard [commit]
    1

    # 重置当前HEAD为指定commit，但保持暂存区和工作区不变

    $ git reset --keep [commit]
    1

    # 新建一个commit 用来撤销指定commit，后者的所有变化都将被前者所抵消，并且应用到当前分支

    $ git revert [commit]
    1

    # 新建一个commit,用来代替当前提交至之前的N次提交，当前提交至之前的N次提交都将被撤销

    $ git revert HEAD~n
    1
